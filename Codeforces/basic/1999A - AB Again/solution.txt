import java.util.Scanner;

public class Solution {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int t = sc.nextInt();
        
        while (t-- > 0) {
            String n = sc.next();
            int sum = (n.charAt(0) - '0') + (n.charAt(1) - '0');
            System.out.println(sum);
        }
        
        sc.close();
    }
}
